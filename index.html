Connect Google Sheets — one-time setup (5 min)

Your app sends trade data to a Google Apps Script URL → the Script appends a row to your Sheet. Also supports reading all rows back for the Dashboard.

1
Open Google Sheets, create a new spreadsheet. Name it PSX Trade Log.
2
Extensions → Apps Script. Delete default code and paste the script below.

} sheet.appendRow([data.date,data.time,data.symbol,data.qty, data.buyPrice,data.sellPrice,data.holdingPeriod, data.grossBuy,data.grossSell, data.brokerageBuy,data.salesTaxBuy,data.leviesBuy,data.totalBuyCost, data.brokerageSell,data.salesTaxSell,data.leviesSell, data.cgtFiler,data.cgtNonFiler, data.totalSellDedFiler,data.totalSellDedNonFiler, data.netProceedsFiler,data.netProceedsNonFiler, data.netProfitFiler,data.netProfitNonFiler, data.returnFiler,data.returnNonFiler]); return ContentService.createTextOutput(JSON.stringify({status:"ok"})).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status:"error",message:err.toString()})).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { if (e.parameter.action === 'getAll') { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var rows = sheet.getDataRange().getValues(); if (rows.length < 2) return ContentService.createTextOutput(JSON.stringify({status:"ok",data:[]})).setMimeType(ContentService.MimeType.JSON); var headers = rows[0]; var keyMap = {"Date":"date","Time":"time","Symbol":"symbol","Qty":"qty","Buy Price":"buyPrice","Sell Price":"sellPrice","Holding Period":"holdingPeriod","Gross Buy":"grossBuy","Gross Sell":"grossSell","Brokerage (Buy)":"brokerageBuy","Sales Tax (Buy)":"salesTaxBuy","Levies (Buy)":"leviesBuy","Total Buy Cost":"totalBuyCost","Brokerage (Sell)":"brokerageSell","Sales Tax (Sell)":"salesTaxSell","Levies (Sell)":"leviesSell","CGT (Filer)":"cgtFiler","CGT (Non-Filer)":"cgtNonFiler","Total Sell Ded (Filer)":"totalSellDedFiler","Total Sell Ded (Non-Filer)":"totalSellDedNonFiler","Net Proceeds (Filer)":"netProceedsFiler","Net Proceeds (Non-Filer)":"netProceedsNonFiler","Net Profit (Filer)":"netProfitFiler","Net Profit (Non-Filer)":"netProfitNonFiler","Return % (Filer)":"returnFiler","Return % (Non-Filer)":"returnNonFiler"}; var data = rows.slice(1).map(function(row){var obj={};headers.forEach(function(h,i){obj[keyMap[h]||h]=row[i];});return obj;}); return ContentService.createTextOutput(JSON.stringify({status:"ok",data:data})).setMimeType(ContentService.MimeType.JSON);
