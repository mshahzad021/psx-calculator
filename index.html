Connect Google Sheets — one-time setup (5 minutes)

Your app sends trade data to a Google Apps Script URL → the Script appends a row to your Sheet automatically. Completely free, no login required in the app.

1
Open Google Sheets, create a new spreadsheet. Name it PSX Trade Log.
2
Go to Extensions → Apps Script. Delete all default code and paste the script below exactly.
function doPost(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); if (sheet.getLastRow() === 0) { sheet.appendRow(["Date","Time","Symbol","Qty","Buy Price","Sell Price", "Holding Period","Gross Buy","Gross Sell", "Brokerage (Buy)","Sales Tax (Buy)","Levies (Buy)","Total Buy Cost", "Brokerage (Sell)","Sales Tax (Sell)","Levies (Sell)", "CGT (Filer)","CGT (Non-Filer)", "Total Sell Ded (Filer)","Total Sell Ded (Non-Filer)", "Net Proceeds (Filer)","Net Proceeds (Non-Filer)", "Net Profit (Filer)","Net Profit (Non-Filer)", "Return % (Filer)","Return % (Non-Filer)"]); } sheet.appendRow([data.date,data.time,data.symbol,data.qty, data.buyPrice,data.sellPrice,data.holdingPeriod, data.grossBuy,data.grossSell, data.brokerageBuy,data.salesTaxBuy,data.leviesBuy,data.totalBuyCost, data.brokerageSell,data.salesTaxSell,data.leviesSell, data.cgtFiler,data.cgtNonFiler, data.totalSellDedFiler,data.totalSellDedNonFiler, data.netProceedsFiler,data.netProceedsNonFiler, data.netProfitFiler,data.netProfitNonFiler, data.returnFiler,data.returnNonFiler]); return ContentService.createTextOutput(JSON.stringify({status:"ok"})).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status:"error",message:err.toString()})).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService.createTextOutput(JSON.stringify({status:"ok",message:"PSX Calculator connected"})).setMimeType(ContentService.MimeType.JSON); }
3
Click Deploy → New deployment. Type: Web app. Execute as: Me. Who has access: Anyone. Click Deploy and authorize.
4
Copy the Web App URL (looks like https://script.google.com/macros/s/ABC.../exec) and paste below.
